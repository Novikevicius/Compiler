int fib(int i)
{ 
    if(i <= 0)
    {
        return 0;
    }
    int c_2 = 0;
    int c_1 = 0;
    int c = 1;
    for(int j = 1; j < i; j += 1)
    {
        c_2 = c_1;
        c_1 = c;
        c = c_1 + c_2;
    }
    return c;
}
